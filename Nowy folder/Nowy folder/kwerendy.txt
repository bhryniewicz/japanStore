Zapytanie 1.

SELECT imie,nazwisko FROM `klienci`
inner join zamowienia on zamowienia.ID_klienta=klienci.ID_klienta
inner join produkty on produkty.ID_Produktu=zamowienia.ID_Produktu
where Grupa_towarowa="notebooki" and producent="HW"


Zapytanie 2.

SELECT DISTINCT(Nazwa_produktu), Grupa_towarowa, Producent FROM `produkty`
inner join zamowienia on produkty.ID_Produktu=zamowienia.ID_Produktu
inner join klienci on klienci.ID_klienta=zamowienia.ID_klienta
where imie="Anna" && nazwisko="Kotnicka"


Zapytanie 3.

SELECT imie,nazwisko, count(ID_zamowienia) as `Liczba_zamowien` FROM `zamowienia` 
inner join produkty on zamowienia.ID_Produktu=produkty.ID_Produktu 
inner join klienci on zamowienia.ID_klienta=klienci.ID_klienta 
group by klienci.ID_klienta
order by Liczba_zamowien desc
limit 1


Zapytanie 4.

SELECT imie,nazwisko, sum(cena_brutto*Liczba_sztuk) FROM `zamowienia` 
inner join produkty on zamowienia.ID_Produktu=produkty.ID_Produktu 
inner join klienci on zamowienia.ID_klienta=klienci.ID_klienta 
group by imie, nazwisko
order by sum(cena_brutto) desc
limit 3


Zapytanie 5.

SELECT sum(Liczba_sztuk) as `Ilosc notebookow kupionych` FROM `zamowienia` 
inner join produkty on zamowienia.ID_Produktu=produkty.ID_Produktu 
inner join klienci on zamowienia.ID_klienta=klienci.ID_klienta 
where Grupa_towarowa="Notebooki" and data BETWEEN '2013-03-01' and '2013-03-31'
